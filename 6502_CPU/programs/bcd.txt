	define INL $02
	define INH $03
	define OUTL $0
	define OUTH $1

	lda #$3F
	sta INL
	lda #$09
	sta INH
	
	lda #0
	sta OUTL
	sta OUTH

	ldx #0

bcd_main_loop:
	clc
	php

	ldy #0
bcd_loop:
	cpy #16
	beq bcd_exit_loop
	plp
	
	rol INL
	rol INH
	rol OUTL
	rol OUTH
	
	lda OUTL
	sec
	sbc #10
	php

	bcc bcd_skip_storing
	STA OUTL
bcd_skip_storing:
	
	iny
	jmp bcd_loop
bcd_exit_loop:
	plp
	
	rol INL
	rol INH
	
	txa
	lsr A
	tay
	lda OUTL
	bcc bcd_skip_store_Hnibble
	asl A
	asl A
	asl A
	asl A
	ora $10,Y
bcd_skip_store_Hnibble:
	sta $10,Y
	inx
	
	lda INH
	ora INL
	beq bcd_exit

	lda #0
	sta OUTH
	sta OUTL
	jmp bcd_main_loop

bcd_exit:
	brk
